# Open an Administrator PowerShell on the production server and run:

# 1. Stop the service (frees .next folder locks)
Stop-Service -Name StickyMyNote -Force

# 2. Navigate to production
cd C:\stick-my-note-prod\stickmynote-client

# 3. Ensure build-time env is in place
Copy-Item .env.local.build-only .env.local -Force

# 4. Build (this may take 5-10 minutes)
pnpm run build

# 5. CRITICAL: Remove .env.local after build
Remove-Item .env.local

# 6. Restart service
Start-Service -Name StickyMyNote

# 7. Verify HTTPS is running
Test-NetConnection -ComputerName localhost -Port 443
